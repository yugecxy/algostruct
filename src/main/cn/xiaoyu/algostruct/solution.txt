public class Solution{
  public int missingNumber(int[] nums) {
     int l = 0;
     int r = nums.length;
     while(l <= r){
        int mid = l + ((r - l) >> 1);
        if(nums[mid] > mid){
            if(mid == 0 || nums[mid-1] == mid - 1){
                return nums[mid];
            }else{
                r = r - 1;
            }
        }else{
           l += 1;
        }
     }
     return -1;
  }
}
