public class Solution{
  public boolean hasCycle(Node node){
      Node fast = node;
      Node slow = node;
      while(fast!=null){
        if(fast.next==null){
          return false;
        }
        fast = fast.next.next;
        slow = slow.next;
        if(fast == slow){
          return true;
        }
      }
      return false;
  }
}
