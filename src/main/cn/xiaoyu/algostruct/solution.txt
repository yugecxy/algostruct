public class Solution{
  public int findRepeatNumber(int[] nums) {
      for(int i = 0;i < nums.length;i++){
        if(i == nums[i]){
          continue;
        }
        if(nums[i] == nums[nums[i]]){
          return nums[i];
        }
        //交换
        int tmp = nums[i];
        nums[i] = nums[nums[i]];
        nums[nums[i]] = tmp;
      }
  }
}
