class Solution {
    public boolean isUgly(int n){
       if(n==1) return true;
       if(n % 2 != 0 && n % 3 !=0 && n % 5 !=0){
         return false;
       }
       boolean ret = true;
       if(n % 2 == 0) ret = ret && isUgly(n/2);
       if(n % 3 == 0) ret = ret && isUgly(n/3);
       if(n % 5 == 0) ret = ret && isUgly(n/5);
       return ret;
    }
}
