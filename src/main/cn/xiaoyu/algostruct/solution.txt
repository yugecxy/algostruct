public class Solution{
  public int binarySearch(int[] arr, int value){
      int left = 0;
      int right = arr.length - 1;
      while(left <= right){
        int mid = left + ((right - left) >> 1);
        if(arr[mid] <= value){
           if(mid == arr.arr.length - 1 || arr[mid+1] > value){
                return arr[mid];
           }
           left = mid + 1;
        }else{
           right = mid - 1;
        }
      }
      return value;
}
