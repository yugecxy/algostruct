public class Solution{
  public double mySqrt(int x) {
     double left = 0;
     double right = x;
     while(left < right){
        double middle = new BigDecimal((left + right)/2).setScale(6,BigDecimal.ROUND_HALF_UP).doubleValue();
        if(middle * middle == x) {
          return middle;
        }
        else if(middle * middle < x){
          left = middle + 0.000001;
        }else{
          right = middle - 0.000001;
        }
     }
     return left;
  }
}
